import { Tabs, Tab } from 'nextra/components'
import { useRouter } from 'next/router'
import { useEffect, useState } from 'react'
import ProChangelog from './changelog/pro-changelog.ja-JP.mdx'
import LocalChangelog from './changelog/local-changelog.ja-JP.mdx'

export const ChangelogTabs = () => {
  const router = useRouter()
  const [activeTab, setActiveTab] = useState(0)

  useEffect(() => {
    const tabParam = router.asPath.split('?tab=')[1]
    const defaultTab = tabParam?.toLowerCase() === 'local' ? 1 : 0
    setActiveTab(defaultTab)
  }, [router.asPath])

  return (
    <Tabs items={['Pro', 'Local']} selectedIndex={activeTab} onChange={setActiveTab}>
      <Tab>
        <ProChangelog/>
      </Tab>
      <Tab>
        <LocalChangelog/>
      </Tab>
    </Tabs>
  )
}

<ChangelogTabs />
